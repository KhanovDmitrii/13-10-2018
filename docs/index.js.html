/**
 Эта функцифя возвращает площадь труегольника, расчитанную по формуле Герона
 @param {number} a - First side of the triangle
 @param {number} b - Second side of the triangle
 @param {number} c - Third side of the triangle
 @constant {number} p - half-meter of the triangle
*/
function sq_ger(a,b,c){
    const p = (a+b+c)/2;
    return Math.sqrt(p*(p-a)*(p-b)*(p-c));
}
